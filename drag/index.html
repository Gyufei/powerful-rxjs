HTML:


      <script src="https://cdn.bootcss.com/rxjs/6.1.0/rxjs.umd.min.js">
      </script>
      <div id="drag">
        <img draggable="false" 
          src="https://i.loli.net/2019/01/11/5c385f53bfbd7.jpg">
      </div>

JavaScript:


      const {
        fromEvent
      } = rxjs

      const {
        concatMapTo,
        takeUntil,
        withLatestFrom
      } = rxjs.operators

      const dragDom = document.getElementById('drag')
      const body = document.body

      const mouseDown = fromEvent(dragDom, 'mousedown')
      const mouseMove = fromEvent(body, 'mousemove')
      const mouseUp = fromEvent(body, 'mouseup')

      mouseDown.pipe(
        concatMapTo(mouseMove.pipe(
          takeUntil(mouseUp)
        )),
        withLatestFrom(mouseDown, (mmove, mdown) => ({
          x: mmove.clientX - mdown.offsetX,
          y: mmove.clientY - mdown.offsetY
        })
        )
      ).subscribe(pos => {
        dragDom.style.transform = `translate(${pos.x}px, ${pos.y}px)`
      })
